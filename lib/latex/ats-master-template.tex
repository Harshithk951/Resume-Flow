% ============================================================
%  ResumeFlow — ATS-STRICT MASTER TEMPLATE (single source of truth)
% ============================================================
%  This is the ONLY resume template in the product. Every job
%  compiles through this exact skeleton. lib/latex/jsonToLatex.ts
%  fills the {{SCALAR}} tokens and emits repeated \resheading /
%  \resitem blocks for array fields (education, experience,
%  projects, certifications).
%
%  Design rules baked in:
%   - Single column, no tables, no icons, no graphics, no text
%     boxes -> parses cleanly in every ATS text extractor.
%   - \skillrow uses a non-breaking space (~) after the colon.
%     THIS IS THE FIX for the "CS FundamentalsAgile/Scrum..."
%     run-on bug — that bug is a missing separator between a
%     bold label and its value, not an AI content problem.
%   - Any \ressection whose underlying array is empty must be
%     OMITTED ENTIRELY by jsonToLatex.ts — never render an empty
%     header (e.g. no "CERTIFICATIONS" heading with nothing under it).
%   - Every {{TOKEN}} value must pass through escapeLatex() before
%     insertion. Never insert raw user/AI text into this template.
% ============================================================

\documentclass[11pt,letterpaper]{article}

\usepackage[T1]{fontenc}
% Computer Modern's default em-dash, en-dash, and "fi"/"fl" ligatures do not
% carry a correct ToUnicode map in pdftex output. Verified by compiling and
% running pdftotext on this exact template: "file-structure" extracted as a
% control character instead of text, silently breaking any ATS keyword match
% on words containing "fi"/"fl" (file, efficient, significant, workflow...).
% expansion=false is required because CM's bitmap/PK fonts aren't scalable —
% microtype's font-expansion feature throws a fatal error without it.
\usepackage[expansion=false]{microtype}
\DisableLigatures{encoding = *, family = *}
\usepackage[left=0.65in,right=0.65in,top=0.55in,bottom=0.55in]{geometry}
\usepackage{enumitem}
\usepackage[hidelinks]{hyperref}

\setlength{\parindent}{0pt}
\pagestyle{empty}

% ---------- Macros (fixed — never regenerate these per template) ----------

% Name / contact
\newcommand{\rname}[1]{{\centering\Huge\bfseries #1\par}}
\newcommand{\rcontact}[1]{{\centering\small #1\par}}

% Section heading with underline rule
\newcommand{\ressection}[1]{%
  \vspace{9pt}
  {\bfseries\large #1}\\[-7pt]
  \noindent\rule{\linewidth}{0.6pt}\vspace{3pt}
}

% Heading row used for education / experience / project entries
% #1 Title or Role   #2 Dates   #3 Org / Institution / Stack   #4 Location (optional, pass {} to omit)
\newcommand{\resheading}[4]{%
  \noindent\textbf{#1} \hfill \textbf{#2}\\[1pt]
  \noindent\textit{#3} \hfill \textit{#4}\\[2pt]
}

% Tight bullet list for accomplishment bullets
\newenvironment{resitems}{%
  \begin{itemize}[leftmargin=16pt,itemsep=1pt,topsep=2pt,parsep=0pt,label=\textbullet]
}{%
  \end{itemize}
}
\newcommand{\resitem}[1]{\item #1}

% Skills row — label: value with guaranteed spacing (bug fix)
\newcommand{\skillrow}[2]{\textbf{#1:}~#2\\[2pt]}

% ============================================================
\begin{document}
