\documentclass[11pt,letterpaper]{article}

\usepackage[T1]{fontenc}
\usepackage[left=0.75in,right=0.75in,top=0.6in,bottom=0.6in]{geometry}
\usepackage{enumitem}
\usepackage[hidelinks]{hyperref}
% Palatino-style serif — traditional, conservative, reads as "trustworthy"
% for finance/banking recruiters. Requires texlive-fonts-recommended.
% If your pdflatex install is minimal (texlive-latex-base only), comment
% out this line and the template falls back cleanly to Computer Modern —
% in which case the microtype fix below becomes REQUIRED, not defensive.
\usepackage{mathpazo}
\linespread{1.02}
% Defensive: mathpazo (Type1 PostScript Palatino clone) tested clean with
% no ligature/dash extraction bug, unlike default Computer Modern. Kept
% here anyway as a guard against the texlive-fonts-recommended fallback
% case above, where this template silently reverts to Computer Modern.
\usepackage[expansion=false]{microtype}
\DisableLigatures{encoding = *, family = *}

\setlength{\parindent}{0pt}
\pagestyle{empty}

\newcommand{\ressection}[1]{%

  {\bfseries\normalsize\textsc{#1}}\\[-6pt]
  \noindent\rule{\linewidth}{0.5pt}\vspace{3pt}
}

\newcommand{\resheading}[4]{%
  \noindent\textbf{#1} \hfill \textbf{#2}\\[1pt]
  \noindent\textit{#3} \hfill \textit{#4}\\[2pt]
}

\newenvironment{resitems}{%
  \begin{itemize}[leftmargin=16pt,itemsep=1pt,topsep=2pt,parsep=0pt,label=\textbullet]
}{%
  \end{itemize}
}
